import spacy
nlp = spacy.load("en_core_web_sm")
text = "SpaCy is a popular natural language processing library."
doc = nlp(text)
for token in doc:
   print(f"Token: {token.text}, POS: {token.pos_}, Lemma: {token.lemma_}")
