from textblob import TextBlob
from textblob.sentiments import NaiveBayesAnalyzer
def get_sentiment(text):
   blob1 = TextBlob(text, analyzer=NaiveBayesAnalyzer())
   sentiment = blob1.sentiment
   return sentiment
input_text ="I love using TextBlob for sentiment analysis. It's great!"
output_sentiment = get_sentiment(input_text)
print(f"Input Text: {input_text}")
print(f"Sentiment: {output_sentiment.classification} (Positive: {output_sentiment.p_pos}, Negative: {output_sentiment.p_neg})")


output:

Input Text: I love using TextBlob for sentiment analysis. It's great!
Sentiment: pos (Positive: 0.9999999071097168, Negative: 9.289028813454466e-08)
